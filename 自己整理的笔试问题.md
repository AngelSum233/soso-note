  function getInter( arr ,  s ) {
      let f = new Array(arr[arr.length-1] + s + 1).fill(0)
      for(let i=1;i<arr.length;i++) {
          if(arr[i] < arr[i-1]) arr[i] = 0
      }
      for(let i=0;i<arr.length;i++) {
          if(arr[i] === 0) continue
          f[arr[i]] = 1
      }
      let n = 0
      for(let i=1;;i++) {

      }
  }
